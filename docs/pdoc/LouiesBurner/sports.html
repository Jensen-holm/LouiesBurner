 92    def _get_baseball_verb(self, stat: str) -> str:
 93        """
 94        Get an appropriate verb for a baseball statistic.
 95
 96        Parameters
 97        ----------
 98        stat : str
 99            The name of the statistic

101        Returns
102        -------

106        Notes
107        -----
108        Returns specific verbs for different types of stats:

126        """
127        Create an engaging tweet about baseball season high achievement(s).
128
129        Parameters

131        highs : list[dict]

158                    "⚡️ RECORD BREAKER ⚡️\n{player} leads the way with {value} {stat_type} against {opponent}! #AnchorUp ⚓️",
159                    "👀 Look what {player} just did!\nNew season high: {value} {stat_type} vs {opponent}! #AnchorUp⚓️",
160                    "💪 BEAST MODE: {player} 💪\nDominates with {value} {stat_type} against {opponent}! #AnchorUp ⚓️",

163                return random.choice(single_templates).format(
164                    player=high["Player"],
165                    value=high["Value"],

194    def get_season_highs_for_date(self, date: datetime.date) -> List[Dict]:
195        """
196        Get baseball season highs that were set/tied on the day before the given date.
197
198        Parameters
199        ----------
200        date : datetime.date
201            The date to check for season highs (will check previous day)
202
203        Returns
204        -------
205        List[Dict]
206            List of dictionaries containing season high information:
207            - Statistic : str
208                Name of the statistic
209            - Value : Any
210                Value achieved
211            - Player : str
212                Player who achieved it
213            - Opponent : str
214                Opponent it was achieved against
215            - Date : datetime.date
216                Date it was achieved
217
218        Notes
219        -----
220        Filters out negative statistics (defined in negative_stats list)
221        and only includes achievements from the previous day.
222        """
223        # Get the previous day's date
224        prev_date = date - datetime.timedelta(days=1)
225
226        # Get the season highs dataframe
227        df = self.season_high_df
228
229        # Track any new or tied season highs
230        new_highs = []
231
232        # Process each row in the dataframe
233        for _, row in df.iterrows():
234            statistic = row["Statistic"]
235
236            # Skip stats we don't want to tweet about
237            if not self._should_tweet_stat(statistic):
238                continue
239
240            high_value = row["High"]
241            players = row["Player"].split("; ")
242            opponents = row["Opponent"].split("; ")
243
244            # Check each player/opponent combination for this statistic
245            for player, opponent in zip(players, opponents):
246                game_date = self._extract_date(opponent)
247
248                # Only include achievements from the previous day
249                if game_date and game_date == prev_date:
250                    new_highs.append(
251                        {
252                            "Statistic": statistic,
253                            "Value": high_value,
254                            "Player": player,
255                            "Opponent": opponent,
256                            "Date": game_date,
257                        }
258                    )
259
260        return new_highs
